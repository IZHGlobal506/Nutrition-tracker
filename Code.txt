<script src="https://cdn.jsdelivr.net/npm/chart.js"></script>
<script>
let foodData = JSON.parse(localStorage.getItem('foodData')) || [];

function renderTable() {
    const tableBody = document.querySelector('#foodTable tbody');
    tableBody.innerHTML = '';
    foodData.forEach((item, index) => {
        const row = `<tr>
            <td>${item.name}</td>
            <td>${item.calories}</td>
            <td>${item.protein}</td>
            <td>${item.carbs}</td>
            <td>${item.fat}</td>
            <td><button onclick="removeFood(${index})">Delete</button></td>
        </tr>`;
        tableBody.innerHTML += row;
    });
    updateChart();
    localStorage.setItem('foodData', JSON.stringify(foodData));
}

function addFood() {
    const name = prompt("Food name:");
    const calories = parseFloat(prompt("Calories:")) || 0;
    const protein = parseFloat(prompt("Protein (g):")) || 0;
    const carbs = parseFloat(prompt("Carbs (g):")) || 0;
    const fat = parseFloat(prompt("Fat (g):")) || 0;

    if (name) {
        foodData.push({ name, calories, protein, carbs, fat });
        renderTable();
    }
}

function removeFood(index) {
    foodData.splice(index, 1);
    renderTable();
}

function clearAll() {
    if (confirm("Clear all data?")) {
        foodData = [];
        renderTable();
    }
}

let chart;
function updateChart() {
    const totalProtein = foodData.reduce((sum, item) => sum + item.protein, 0);
    const totalCarbs = foodData.reduce((sum, item) => sum + item.carbs, 0);
    const totalFat = foodData.reduce((sum, item) => sum + item.fat, 0);

    const ctx = document.getElementById('macroChart').getContext('2d');
    if (chart) chart.destroy();
    chart = new Chart(ctx, {
        type: 'pie',
        data: {
            labels: ['Protein', 'Carbs', 'Fat'],
            datasets: [{
                data: [totalProtein, totalCarbs, totalFat],
                backgroundColor: ['#4CAF50', '#2196F3', '#FF9800']
            }]
        }
    });
}

renderTable();
</script>
